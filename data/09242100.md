``` mermaid
flowchart TD
    A[프로그램 시작] --> B[바이너리 파일 읽기]
    B --> C[메뉴 출력]

    %% 메뉴 선택
    C -->|1 List| D[회원 목록 출력]
    D --> C

    C -->|2 Add| E[회원 추가]
    E --> F{전화번호 중복?}
    F -->|중복| C
    F -->|중복 아님| G[회원 저장 후 메시지 출력]
    G --> C

    C -->|3 Edit| H[회원 이름 입력]
    H --> I{이름 유효성 검사 통과?}
    I -->|False| C
    I -->|True| J{회원 존재?}
    J -->|없음| C
    J -->|있음| K[회원 목록 출력]
    K --> L[수정할 번호 선택]
    L --> M{번호 유효성 검사}
    M -->|False| K
    M -->|True| N[항목 입력 및 유효성 검사]
    N --> O{전화번호 중복?}
    O -->|중복| K
    O -->|중복 아님| P[수정 완료 메시지 출력]
    P --> C

    C -->|4 Delete| Q[회원 이름 입력]
    Q --> R{이름 유효성 검사 통과?}
    R -->|False| C
    R -->|True| S{회원 존재?}
    S -->|없음| C
    S -->|True| T[회원 목록 출력]
    T --> U[삭제할 번호 입력]
    U --> V{번호 유효성 검사}
    V -->|False| T
    V -->|True| W[삭제 확인 메시지 출력]
    W -->|1 삭제| X[목록에서 삭제 후 메시지 출력]
    X --> C
    W -->|2 취소| C

    C -->|5 Exit| Y[종료 확인 메시지]
    Y -->|1 True| Z[목록 저장 후 프로그램 종료]
    Y -->|2 False| C
```
