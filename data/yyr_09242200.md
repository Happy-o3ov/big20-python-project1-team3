```mermaid
flowchart TD
    Start([프로그램 시작])
    LoadFile[바이너리 파일 읽기]
    Menu[메뉴 출력 및 입력 (1~5)]
    
    Menu -->|1| List[목록 출력]
    Menu -->|2| AddStart[회원 추가 시작]
    Menu -->|3| ModStart[회원 수정 시작]
    Menu -->|4| DelStart[회원 삭제 시작]
    Menu -->|5| ExitConfirm[종료 확인]
    Menu -->|기타| MenuError[올바른 번호 입력 요청]

    MenuError --> Menu

    %% 목록 출력
    List --> DisplayList[순번, 이름, 전화번호, 관계, 주소, 총 개수 출력]
    DisplayList --> Menu

    %% 회원 추가
    AddStart --> AddName[이름 입력 및 유효성 검사<br/>한영 5자 이내, 필수]
    AddNameValid{유효한가?}
    AddName --> AddNameValid
    AddNameValid -- 아니오 --> AddErrorName[이름 오류 메시지 출력] --> Menu
    AddNameValid -- 예 --> AddPhone[전화번호 입력 및 유효성 검사<br/>000-000-000 형식, 숫자와 -만, 필수]
    AddPhoneValid{유효한가?}
    AddPhone --> AddPhoneValid
    AddPhoneValid -- 아니오 --> AddErrorPhone[전화번호 오류 메시지 출력] --> Menu
    AddPhoneValid -- 예 --> AddRel[관계 입력 및 유효성 검사<br/>1,2,3 중 선택 필수]
    AddRelValid{유효한가?}
    AddRel --> AddRelValid
    AddRelValid -- 아니오 --> AddErrorRel[관계 오류 메시지 출력] --> Menu
    AddRelValid -- 예 --> AddAddr[주소 입력 (100자 이내)]
    AddAddrValid{주소 길이 100자 이내?}
    AddAddr --> AddAddrValid
    AddAddrValid -- 아니오 --> AddErrorAddr[주소 오류 메시지 출력] --> Menu
    AddAddrValid -- 예 --> CheckDupPhoneAdd[전화번호 중복 검사]
    CheckDupPhoneAddDup{중복인가?}
    CheckDupPhoneAdd --> CheckDupPhoneAddDup
    CheckDupPhoneAddDup -- 예 --> AddDupMsg[기존 전화번호 있음 메시지 출력] --> Menu
    CheckDupPhoneAddDup -- 아니오 --> SaveAdd[회원 정보 저장]
    SaveAddDone[저장 완료 메시지 출력]
    SaveAdd --> SaveAddDone --> Menu

    %% 회원 수정
    ModStart --> ModName[수정할 회원 이름 입력 및 유효성 검사<br/>한영 1~5자]
    ModNameValid{유효한가?}
    ModName --> ModNameValid
    ModNameValid -- 아니오 --> ModErrorName[이름 오류 메시지 출력] --> Menu
    ModNameValid -- 예 --> SearchName[이름으로 회원 검색]
    SearchFound{데이터 있음?}
    SearchName --> SearchFound
    SearchFound -- 아니오 --> NotFoundMsg[없습니다 메시지 출력] --> Menu
    SearchFound -- 예 --> DisplayModList[목록 출력: 순번, 이름, 전화번호, 관계, 주소, 총 개수]
    DisplayModList --> SelectModNo[수정할 번호 입력 및 유효성 검사]
    SelectModValid{번호 유효한가?}
    SelectModNo --> SelectModValid
    SelectModValid -- 아니오 --> ModNoError[번호 오류 메시지 출력] --> DisplayModList
    SelectModValid -- 예 --> ModInputName[수정할 이름 입력 및 유효성 검사]
    ModInputNameValid{유효한가?}
    ModInputName --> ModInputNameValid
    ModInputNameValid -- 아니오 --> ModInputNameError[이름 오류 메시지 출력] --> DisplayModList
    ModInputNameValid -- 예 --> ModInputPhone[수정할 전화번호 입력 및 유효성 검사]
    ModInputPhoneValid{유효한가?}
    ModInputPhone --> ModInputPhoneValid
    ModInputPhoneValid -- 아니오 --> ModInputPhoneError[전화번호 오류 메시지 출력] --> DisplayModList
    ModInputPhoneValid -- 예 --> CheckDupPhoneMod[전화번호 중복 검사]
    CheckDupPhoneModDup{중복인가?}
    CheckDupPhoneMod --> CheckDupPhoneModDup
    CheckDupPhoneModDup -- 예 --> ModDupMsg[기등록 전화번호 중복 메시지 출력] --> DisplayModList
    CheckDupPhoneModDup -- 아니오 --> ModInputRel[관계 입력 및 유효성 검사]
    ModInputRelValid{유효한가?}
    ModInputRel --> ModInputRelValid
    ModInputRelValid -- 아니오 --> ModInputRelError[관계 오류 메시지 출력] --> DisplayModList
    ModInputRelValid -- 예 --> ModInputAddr[주소 입력 (100자 이내) 및 검사]
    ModInputAddrValid{주소 길이 100자 이내?}
    ModInputAddr --> ModInputAddrValid
    ModInputAddrValid -- 아니오 --> ModInputAddrError[주소 오류 메시지 출력] --> DisplayModList
    ModInputAddrValid -- 예 --> SaveMod[수정 저장]
    SaveModDone[수정 완료 메시지 출력]
    SaveMod --> SaveModDone --> Menu

    %% 회원 삭제
    DelStart --> DelName[삭제할 회원 이름 입력 및 유효성 검사<br/>한영 1~5자]
    DelNameValid{유효한가?}
    DelName --> DelNameValid
    DelNameValid -- 아니오 --> DelErrorName[이름 오류 메시지 출력] --> Menu
    DelNameValid -- 예 --> SearchDel[이름으로 회원 검색]
    SearchDelFound{데이터 있음?}
    SearchDel --> SearchDelFound
    SearchDelFound -- 아니오 --> DelNotFoundMsg[없습니다 메시지 출력] --> Menu
    SearchDelFound -- 예 --> DisplayDelList[목록 출력: 순번, 이름, 전화번호, 관계, 주소, 총 개수]
    DisplayDelList --> SelectDelNo[삭제할 번호 입력 및 유효성 검사]
    SelectDelValid{번호 유효한가?}
    SelectDelNo --> SelectDelValid
    SelectDelValid -- 아니오 --> DelNoError[번호 오류 메시지 출력] --> DisplayDelList
    SelectDelValid -- 예 --> ConfirmDel[삭제 확인 메시지 출력 및 입력 (1예 2취소)]
    ConfirmDelChoice{삭제할까?}
    ConfirmDel --> ConfirmDelChoice
    ConfirmDelChoice -- 1 --> Delete[목록에서 삭제]
    DeleteDone[삭제 완료 메시지 출력]
    Delete --> DeleteDone --> Menu
    ConfirmDelChoice -- 2 --> Menu

    %% 종료
    ExitConfirm --> ExitChoice[종료 확인 메시지 및 입력 (1예 2아니오)]
    ExitChoiceConfirm{종료할까?}
    ExitChoice --> ExitChoiceConfirm
    ExitChoiceConfirm -- 1 --> SaveFile[회원 목록 파일 저장 및 종료]
    ExitChoiceConfirm -- 2 --> Menu

    %% 연결 시작
    Start --> LoadFile --> Menu
```
